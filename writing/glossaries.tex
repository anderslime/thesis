\newacronym{odg}{ODG}{Object Dependence Graph}
\newacronym{idg}{IDG}{Instance Dependence Graph}
\newacronym{ddg}{DDG}{Declaration Dependence Graph}
\newacronym{dup}{DUP}{Data Update Propagation}
\newacronym{cms}{CMS}{Content Management System}

\newglossaryentry{object-dependence-graph}{
  name={Object Dependence Graph},
  description={A data structure to represent dependencies between cached objects and underlying data.}
}

\newglossaryentry{instance-dependence-graph}{
  name={Instance Dependence Graph},
  description={A data structure to represent the dynamic dependencies between cached objects and underlying data.}
}

\newglossaryentry{declaration-depenence-graph}{
  name={Declaration Dependence Graph},
  description={A data structure to represent the static dependencies between cached objects and underlying data.}
}

\newglossaryentry{data-update-prop}{
  name={Data Update Propagation},
  description={The procedure of propagating updates to affected data based on a given event.}
}

\newglossaryentry{content-man-system}{
  name={Content Management System},
  description={A system centered around users being able to edit content.}
}

\newglossaryentry{underlying-data}{
  name={underlying data},
  description={The data used as input for a computation such that the result of the computation changes when the underlying data changes.}
}

\newglossaryentry{cached-object}{
  name={cached object},
  description={An abstraction of an object being cached, which includes the value and metadata used in the cache system.}
}
